SET NAMES utf8mb4;

-- 1) عمود reference في stock_movements (مخزون الطلبات / الحجز) — يُضاف فقط إن لم يكن موجوداً
SET @col_exists := (
    SELECT COUNT(*) FROM INFORMATION_SCHEMA.COLUMNS
    WHERE TABLE_SCHEMA = DATABASE()
      AND TABLE_NAME = 'stock_movements'
      AND COLUMN_NAME = 'reference'
);
SET @sql := IF(
    @col_exists > 0,
    'SELECT ''stock_movements.reference already exists'' AS msg',
    'ALTER TABLE stock_movements ADD COLUMN reference VARCHAR(100) NULL'
);
PREPARE stmt FROM @sql;
EXECUTE stmt;
DEALLOCATE PREPARE stmt;

-- 2) ربط كل المنتجات بكل القنوات النشطة (لا يكرر الصفوف الموجودة)
INSERT IGNORE INTO product_channels (product_id, channel_id)
SELECT p.id, c.id
FROM products p
CROSS JOIN channels c
WHERE c.is_active = 1;
